━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 STEP 1: GET API KEY FROM GOOGLE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Go to: https://console.cloud.google.com/

2. Sign in with Gmail

3. Create New Project:
   ┌─────────────────────────────────────┐
   │ Click: "NEW PROJECT"                │
   │ Name: ACU Project Map               │
   │ Click: "CREATE"                     │
   └─────────────────────────────────────┘

4. Enable APIs (do this twice):
   ┌─────────────────────────────────────┐
   │ Go to: APIs & Services → Library    │
   │                                     │
   │ Search: "Maps JavaScript API"       │
   │ Click: ENABLE                       │
   │                                     │
   │ Search: "Places API"                │
   │ Click: ENABLE                       │
   └─────────────────────────────────────┘

5. Create API Key:
   ┌─────────────────────────────────────┐
   │ Go to: APIs & Services → Credentials│
   │ Click: "+ CREATE CREDENTIALS"       │
   │ Select: "API key"                   │
   │ Click: COPY (copy your key!)        │
   └─────────────────────────────────────┘

   Your key looks like:
   AIzaSyB1234567890abcdefghijklmnopqrstuvwxyz


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 STEP 2: INSTALL GOOGLE MAPS LIBRARY
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

EASIEST WAY:
   ┌─────────────────────────────────────┐
   │ Double-click:                       │
   │ install-dependencies.bat            │
   └─────────────────────────────────────┘

OR use Command Prompt:
   ┌─────────────────────────────────────┐
   │ Win+R → type "cmd" → Enter          │
   │                                     │
   │ cd "C:\Users\edujk\Desktop\project map\acu-project-map-123456-789"
   │                                     │
   │ npm install @react-google-maps/api @types/google.maps
   └─────────────────────────────────────┘


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 STEP 3: CREATE .env FILE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Go to project folder in File Explorer

2. Right-click → New → Text Document

3. Name it: .env (remove .txt extension)

4. Open with Notepad

5. Type this (with YOUR key):
   ┌─────────────────────────────────────┐
   │ VITE_GOOGLE_MAPS_API_KEY=AIzaSy...  │
   └─────────────────────────────────────┘

6. Save and close

⚠️  IMPORTANT: 
    - File must be named exactly: .env
    - Must be in project root folder
    - No spaces, no quotes


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 STEP 4: RUN YOUR APP
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Stop server (Ctrl+C)

2. Start again:
   ┌─────────────────────────────────────┐
   │ npm run dev                         │
   └─────────────────────────────────────┘

3. Open browser:
   ┌─────────────────────────────────────┐
   │ http://localhost:8080/              │
   └─────────────────────────────────────┘

4. ✅ You should see Google Maps!

❌ "API Key Missing" error:
   → Check .env file exists
   → Check .env file format (no spaces/quotes)
   → Restart dev server

❌ Map not loading:
   → Check browser console (F12)
   → Verify APIs enabled in Google Cloud
   → Wait 5 minutes for changes to propagate

❌ PowerShell blocked:
   → Use Command Prompt instead (Win+R → cmd)
   → Or use install-dependencies.bat
